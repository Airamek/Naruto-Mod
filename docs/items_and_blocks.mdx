# Items & Blocks

export const scale = 2.12;

export const BackgroundImg = ({clazz, style}) => (
    <div className={clazz} style={{
            background: 'url(/img/mc/gui/crafting_table.png)',
            imageRendering: 'pixelated',
            backgroundSize: (256 * scale),
            ...style
    }}></div>
);

export function capitaliseName(name = '') {
    return name.split('_').map(word => word.charAt(0).toUpperCase() + word.slice(1)).join(' ');
}

export const ItemRender = ({item, style, count}) => (item ?
    <div className={`item-render`} style={{
            background: item ? `url(/img/mc/item/${item}.png)` : undefined,
            backgroundSize: (16 * scale),
            width: (16 * scale),
            height: (16 * scale),
            imageRendering: 'pixelated',
            ...style
    }} title={capitaliseName(item) + (count ? ` (${count})` : '')}>{count && <div className="mc-text" style={{
        position: 'absolute',
        transform: 'translateX(-50%)',
        filter: `drop-shadow(${1 * scale}px ${1 * scale}px 0px #0009)`,
        fontSize: (9 * scale),
        color: '#fff',
        left: (14 * scale),
        top: (7 * scale),
    }}>
        {count}
    </div>}</div> : <div style={{
            width: (16 * scale),
            height: (16 * scale),
            ...style
        }}></div>
);

export const RowRender = ({row, keyMap}) => {
    return <>
        {row && row.split('').map((item, colIndex) => {
            return <ItemRender key={colIndex} item={keyMap[item]} style={{
            margin: (1 * scale),
        }}/>;
        })}
    </>
}

export const ItemGrid = ({keyMap, grid, style}) => {
    return (<div style={{
        ...style
    }}>
        {grid && grid.map((row, rowIndex) => (<div key={rowIndex} style={{
            display: 'flex',
            flexDirection: 'row',
            flexWrap: 'wrap',
            justifyContent: 'center',
            alignItems: 'center',
        }}>
            <RowRender keyMap={keyMap} row={row} />
        </div>))}
    </div>)
}

export const CraftingRecipe = ({keyMap = {}, grid, result, count}) => {
    return <div style={{
        position: 'relative',
        width: (146 * scale),
        height: (72 * scale)
    }}>
        <BackgroundImg className="top" style={{
            width: (143 * scale),
            height: (3 * scale),
        }}/>
        <BackgroundImg className="center" style={{
            marginLeft: (3 * scale),
            width: (140 * scale),
            height: (65 * scale),
            backgroundPositionX: (-20 * scale),
            backgroundPositionY: (-9 * scale),
        }}/>
        <BackgroundImg className="bottom" style={{
            width: (143 * scale),
            height: (4 * scale),
            backgroundPositionY: (-162 * scale),
        }}/>
        <BackgroundImg className="left" style={{
            position: 'absolute',
            width: (3 * scale),
            height: (65 * scale),
            backgroundPositionY: (-3 * scale),
            top: (3 * scale),
        }}/>
        <BackgroundImg className="right" style={{
            position: 'absolute',
            width: (3 * scale),
            height: (69 * scale),
            backgroundPositionX: (-173 * scale),
            right: (0 * scale),
            top: 0,
        }}/>
        <BackgroundImg className="bottomright" style={{
            position: 'absolute',
            width: (3 * scale),
            height: (3 * scale),
            backgroundPositionX: (-173 * scale),
            backgroundPositionY: (-163 * scale),
            right: (0 * scale),
            top: (69 * scale),
        }}/>
        <ItemGrid keyMap={keyMap} grid={grid} style={{
            position: 'absolute',
            left: (12 * scale),
            top: (10 * scale),
        }}></ItemGrid>
        <ItemRender item={result} style={{
            position: 'absolute',
            left: (107 * scale),
            top: (29 * scale),
        }} count={count}/>
    </div>
};

export const ItemView = ({src}) => <img src={src}
    width="100"
    style={{
        imageRendering: 'pixelated',
    }}/>;

Crafting recipes can be found in game in the vanilla crafting recipes menu in the crafting bench and can be unlocked by finding the required materials.

If you want to see all in game by default you can also install [JEI](https://www.curseforge.com/minecraft/mc-mods/jei). If you want to search all items and blocks relating to the naruto mod you can search `@narutomod`.
